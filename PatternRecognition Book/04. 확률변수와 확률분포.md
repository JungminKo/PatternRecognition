## 가우시안 확률밀도함수값 구하기
````matlab
function [z] = gausspdf(x, mu, sigma);
% GAUSSPDF 가우시안 확률밀도함수값 구하기
%   
%   GAUSSPDF(X, MU, SIGMA)는 평균 MU와 공분산 SIGMA에 따른
%   가우시안 프로세스에 따른 데이터 X 혹은 데이터의 우도를 리턴.
%   MU는 1*D 벡터(여기서 D는 프로세스의 차원)
%   SIGMA는 D*D 행렬
%   X는 N*D 행렬로 각 행이 하나의 D차원 데이터 점이 된다.

[N, D] = size(x);
if (min(N, D) ==1), x=x(:); end;

invSig = inv(sigma);
mu = mu(:);

x = xrepmat(mu, N, 1);

z = sum( ( (x*invSig).*x), 2);

z = exp(-0.5*z) / sqrt((2*pi)^D*det(sigma));
````

## 가우시안 컨투어 그리기
````matlab
function [h] = plotguas(mu, sigma, colspec);

% PLOTGAUS 가우시안 컨투어 그리기
% PLOTGAUS(MU, SIGMA, COLSPEC)
% MU : 평균
% SIGMA : 공분산
% 색깔지정인자 COLSPEC = [R, G, B].
% PLOTGAUS(MU, SIGMA)일 경우, 디폴트 색상은 [0, 0, 1] (cyan).
% 

if margin<3; colspec = [0, 0, 1]; end;
npts = 100;

stdev = sqrtm(sigma);

t = linspace(-pi, pi, npts);
t = t(:);

X = [cos(t) sin(t)] * stdev + repmat(mu, npts, 1);

h(1) = line(X(:, 2), 'color', colspec, 'linew', 2);
h(2) = line(mu(1), mu(2), 'marker', '+', 'markersize', 10, 'color', colspec, 'linew', 2);
````
